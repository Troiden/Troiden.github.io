Get version info:

function GetAppVersion: String;
var
Res: TResourceStream;
VerInfo: record Minor, Major, Build, Release: Word end;
begin
Res := TResourceStream.Create(HInstance, '#1', RT_VERSION);
try
Res.Seek(48, soBeginning);
Res.Read(VerInfo, SizeOf(VerInfo));
Result := Format('%d.%d.%d.%d', [VerInfo.Major, VerInfo.Minor, VerInfo.Release, VerInfo.Build]);
finally
Res.Free;
end;
end;
